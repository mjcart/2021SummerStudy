import java.util.Scanner;

public class Q2440 {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);
    int n =sc.nextInt();
    sc.close();
    for(int i=1;i<=n;i++){
      for(int j=0;j<=n-i;j++){
        System.out.print("*");
      }
      System.out.println();
    }
    
	}
}
